После:
Устанавливаем ng_car, он находится в /usr/ports/net/ng_car

Устанавливаем mpd4:

 

cd /usr/ports/net/mpd4

make config install clean  - ставим галочку на против

 

NG_CAR                   Use ng_car kernel module from port (< 7.0 only)  (устанавливает ng_car из портов для FreeBSD ранее седьмой версии, в версиях 7.0 ng_car, присутствует в модулях ядра). 
NG_IPACCT              Use ng_ipacct kernel module from port (для учета трафика на интерфейсах ng)

 

Конфиги mpd:

 

Будьте внимательны со знаками табуляции и пробелами. Пробелы нельзя использовать для выделения.

 

 default:
(табуляция)load pptp0
pptp0:
(Табуляция)new -i ng0 pptp0 pptp0

 mpd.conf

#Секция логов
startup:
    log ccp
    log ccp2
    log ccp3
    log chat2
    log echo
    log frame
    log mp
    log mp2
    log pptp0
##########
    log +radius +radius2 # В логе mpd будет отображаться его общение с radius-ом

 

default:
    load pptp0

 

pptp0:
    new -i ng0 pptp0 pptp0
    set ipcp ranges 10.50.0.1/32 10.20.5.0/24
    set ipcp dns 10.20.3.30
    set pptp self 10.50.0.1
    load client_standart

 

client_standart:
    set iface up-script /usr/local/etc/mpd4/up-down_script/up_script
    set iface down-script /usr/local/etc/mpd4/up-down_script/down_script
    set iface disable on-demand
    set iface enable tee ipacct 
    set bundle disable multilink
    set link yes acfcomp protocomp
#требуем chap авторизацию

 

    set link no pap chap
    set link enable chap-msv2
    set link keep-alive 60 180
    set ipcp yes vjcomp

#чтобы комп видел корпоративную сеть
    set iface enable proxy-arp

 

#enable compression
    set bundle enable compression
    set ccp yes mppc
    set ccp yes mpp-e40
    set ccp yes mpp-e56
    set ccp yes mpp-e128
    set ccp yes mpp-stateless
    set ccp yes mppe-policy
    set pptp enable incoming
    set pptp disable originate
    set bundle yes crypt-reqd
    set iface mtu 1400
    set link mtu 1400

 

#Включение авторизации и аккаунтинга через radius
    set auth enable radius-auth
    set auth enable radius-acct
##########
    set radius server 127.0.0.1 1111 1812 1813
    set radius timeout 10

 

#Конфиг mpd для радиуса
    set radius config /usr/local/etc/mpd4/radius.conf
    set radius retries 5
    set radius enable message-authentic 

 

/usr/local/etc/mpd4/radius.conf

acct 127.0.0.1 1111 3 2
auth 127.0.0.1 1111 3 2

   set link type pptp
